package com.oceanverse.ai.prompt;

import org.springframework.stereotype.Component;
import java.util.Map;

@Component
public class PromptTemplateManager {

    private static final Map<String, String> TEMPLATES = Map.of(
        "SPECIES", SPECIES_PROMPT,
        "ECOLOGY", ECOLOGY_PROMPT,
        "CONSERVATION", CONSERVATION_PROMPT,
        "GENERAL", GENERAL_PROMPT
    );

    /**
     * 根据问题类型获取对应的 System Prompt
     */
    public String getSystemPrompt(String questionType) {
        return TEMPLATES.getOrDefault(questionType, GENERAL_PROMPT);
    }

    // ========== Prompt 模板定义 ==========

        你是一位海洋生物分类学专家，精通全球 23 万种已知海洋生物的形态特征、分类学地位和分布规律。

## 任务 2.2：RAG 检索增强问答

### 背景

裸模型对海洋专业知识的回答可能产生"幻觉"。通过 RAG（Retrieval-Augmented Generation）架构，先从项目自有数据（物种表、观测记录）中检索相关片段，再作为上下文送入模型，可以显著提升回答的准确性和可溯源性。

### 实施步骤

**新建目录**：`smart-ai-service/src/main/java/com/oceanverse/ai/rag/`

**新建文件 1**：`DocumentLoader.java`（从数据库加载知识文档）

```java
package com.oceanverse.ai.rag;

import com.oceanverse.ai.mapper.SpeciesMapper;
import com.oceanverse.pojo.entity.Species;
import lombok.RequiredArgsConstructor;
import org.springframework.ai.document.Document;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

@Component
@RequiredArgsConstructor
public class DocumentLoader {

    private final SpeciesMapper speciesMapper;

    /**
     * 从 species 表加载所有物种数据，转为 Document 列表
     */
    public List<Document> loadSpeciesDocuments() {
        List<Species> speciesList = speciesMapper.selectList(null);
        
        return speciesList.stream()
            .map(species -> {
                String content = String.format("""
                    物种名称: %s
                    学名: %s
                    分类: %s / %s / %s
                    分布区域: %s
                    栖息深度: %s
                    保护等级: %s
                    描述: %s
                    """,
                    species.getCommonName(),
                    species.getScientificName(),
                    species.getPhylum(), species.getClassName(), species.getOrderName(),
                    species.getDistribution(),
                    species.getHabitatDepth(),
                    species.getConservationStatus(),
                    species.getDescription()
                );
                
                return new Document(content);
            })
            .collect(Collectors.toList());
    }
}
```

**新建文件 2**：`KnowledgeBaseService.java`（向量化 + 检索）

```java
package com.oceanverse.ai.rag;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.ai.document.Document;
import org.springframework.ai.embedding.EmbeddingModel;
import org.springframework.ai.vectorstore.SearchRequest;
import org.springframework.ai.vectorstore.VectorStore;
import org.springframework.stereotype.Service;

import jakarta.annotation.PostConstruct;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class KnowledgeBaseService {

    private final DocumentLoader documentLoader;
    private final EmbeddingModel embeddingModel;
    private final VectorStore vectorStore;  // 需要在配置类中创建 RedisVectorStore Bean

    /**
     * 应用启动时初始化知识库
     */
    @PostConstruct
    public void initKnowledgeBase() {
        log.info("开始初始化海洋知识库...");
        List<Document> documents = documentLoader.loadSpeciesDocuments();
        vectorStore.add(documents);
        log.info("知识库初始化完成，共加载 {} 个文档", documents.size());
    }

    /**
     * 根据用户问题检索相关知识
     * @param query 用户问题
     * @param topK 返回最相关的 K 个文档
     * @return 相关文档列表
     */
    public List<Document> search(String query, int topK) {
        return vectorStore.similaritySearch(
            SearchRequest.query(query).withTopK(topK).withSimilarityThreshold(0.7)
        );
    }

    /**
     * 将检索结果格式化为上下文文本
     */
    public String formatContext(List<Document> documents) {
        StringBuilder sb = new StringBuilder("以下是相关参考资料：\n\n");
        for (int i = 0; i < documents.size(); i++) {
            sb.append(String.format("【资料 %d】\n%s\n\n", i + 1, documents.get(i).getContent()));
        }
        return sb.toString();
    }
}
```

package com.oceanverse.ai.config;

import org.springframework.ai.embedding.EmbeddingModel;
import org.springframework.ai.vectorstore.RedisVectorStore;
import org.springframework.ai.vectorstore.VectorStore;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.connection.RedisConnectionFactory;

@Configuration
public class RedisVectorStoreConfig {

    @Value("${spring.ai.vectorstore.redis.index:ai:knowledge:index}")
    private String redisIndex;

    @Value("${spring.ai.vectorstore.redis.prefix:ai:knowledge:doc:}")
    private String redisPrefix;

    @Bean
    public VectorStore vectorStore(RedisConnectionFactory connectionFactory,
                                   EmbeddingModel embeddingModel) {
        return RedisVectorStore.builder(connectionFactory, embeddingModel)
                .indexName(redisIndex)
                .prefix(redisPrefix)
                .build();
    }
}

### 关键技术点

1. **Embedding 向量化**：`text-embedding-v3` 将文本转为高维向量，语义相似的文本在向量空间中距离更近
2. **Redis 向量存储**：Redis Stack 的 RediSearch 模块支持向量索引和相似度检索
3. **检索阈值**：`withSimilarityThreshold(0.7)` 过滤掉相关性低的文档，避免噪声干扰
4. **上下文注入**：将检索结果拼入 System Prompt，让模型基于真实数据回答
5. **来源引用**：可以在回答中要求模型标注引用来源（资料 1 / 资料 2）

## 任务 2.3：多轮对话上下文管理

### 背景

当前 `ChatDTO` 只有 `question`，没有历史消息。每次提问都是独立的，无法实现连贯对话。需要基于 `sessionId` 缓存最近 N 轮对话。

### 实施步骤

**新建文件**：`smart-ai-service/src/main/java/com/oceanverse/ai/session/ChatSessionManager.java`

```java
package com.oceanverse.ai.session;

import lombok.RequiredArgsConstructor;
import org.springframework.ai.chat.messages.AssistantMessage;
import org.springframework.ai.chat.messages.Message;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.TimeUnit;

@Component
@RequiredArgsConstructor
public class ChatSessionManager {

    private static final String SESSION_KEY_PREFIX = "ai:session:";
    private static final long SESSION_TTL_MINUTES = 30;

    private final RedisTemplate<String, String> redisTemplate;
    private final com.fasterxml.jackson.databind.ObjectMapper objectMapper;
    private final com.oceanverse.ai.config.AiProperties aiProperties;

    /**
     * 保存用户提问和 AI 回答到会话历史
     */
    public void saveMessage(String sessionId, String userQuestion, String aiAnswer) {
        String key = SESSION_KEY_PREFIX + sessionId;
        
        try {
            // 构造消息对
            SessionMessage msg = new SessionMessage(userQuestion, aiAnswer);
            String json = objectMapper.writeValueAsString(msg);
            
            // 追加到 Redis List
            redisTemplate.opsForList().rightPush(key, json);
            
            // 只保留最近 N 轮（N = historyRounds * 2，因为每轮有 user + assistant 两条）
            int maxMessages = aiProperties.getHistoryRounds() * 2;
            redisTemplate.opsForList().trim(key, -maxMessages, -1);
            
            // 刷新 TTL
            redisTemplate.expire(key, SESSION_TTL_MINUTES, TimeUnit.MINUTES);
            
        } catch (Exception e) {
            // 会话保存失败不影响主流程
        }
    }

    /**
     * 获取会话历史（用于构造多轮对话上下文）
     */
    public List<Message> getHistoryMessages(String sessionId) {
        String key = SESSION_KEY_PREFIX + sessionId;
        List<String> messages = redisTemplate.opsForList().range(key, 0, -1);
        
        if (messages == null || messages.isEmpty()) {
            return new ArrayList<>();
        }

        List<Message> result = new ArrayList<>();
        try {
            for (String json : messages) {
                SessionMessage msg = objectMapper.readValue(json, SessionMessage.class);
                result.add(new UserMessage(msg.getUserQuestion()));
                result.add(new AssistantMessage(msg.getAiAnswer()));
            }
        } catch (Exception e) {
            // 解析失败返回空列表
        }
        
        return result;
    }

    /**
     * 清空会话历史
     */
    public void clearSession(String sessionId) {
        redisTemplate.delete(SESSION_KEY_PREFIX + sessionId);
    }

    // 内部 DTO
    private record SessionMessage(String userQuestion, String aiAnswer) {}
}
```

**修改 AiServiceImpl.chatStream()**：集成多轮对话

```java
@Override
public void chatStream(ChatDTO dto, HttpServletResponse response) {
    // ... 前置代码 ...

    // 1. 获取会话历史
    List<Message> historyMessages = chatSessionManager.getHistoryMessages(dto.getSessionId());

    // 2. 构造消息列表（System + History + Current）
    List<Message> messages = new ArrayList<>();
    messages.add(new SystemMessage(finalSystemPrompt));  // 来自 Prompt 模板 + RAG
    messages.addAll(historyMessages);                     // 历史对话
    messages.add(new UserMessage(dto.getQuestion()));     // 当前提问

    // 3. 流式调用
    StringBuilder fullAnswer = new StringBuilder();
    
    chatClient.prompt()
        .messages(messages)
        .stream()
        .content()
        .doOnNext(chunk -> {
            // ... 写入 SSE ...
            fullAnswer.append(chunk);
        })
        .doOnComplete(() -> {
            // 保存会话历史
            chatSessionManager.saveMessage(
                dto.getSessionId(), 
                dto.getQuestion(), 
                fullAnswer.toString()
            );
            // ... 保存数据库、发布事件 ...
        })
        .subscribe();
}
```

### 关键技术点

1. **Redis List 存储**：每个 sessionId 对应一个 List，按时间顺序追加消息
2. **窗口裁剪**：`trim(key, -maxMessages, -1)` 只保留最近 N 条，避免上下文过长
3. **TTL 自动过期**：30 分钟无活动后会话自动清除，节省内存
4. **Spring AI Message 抽象**：`UserMessage` / `AssistantMessage` / `SystemMessage` 统一表示对话消息

### 验收标准

- [ ] 同一个 sessionId 连续提问"绿海龟吃什么？" → "它生活在哪些海域？"，AI 能理解"它"指绿海龟
- [ ] 超过 10 轮对话后，最早的历史被自动裁剪
- [ ] 30 分钟无活动后，Redis 中的会话 key 自动过期
- [ ] 新建一个 sessionId 时，历史为空，AI 正常回答
